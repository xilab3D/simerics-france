<script>
(function() {
  const buttons = document.querySelectorAll('.filter-btn');
  const items = document.querySelectorAll('.gallery-item');
  const empty = document.querySelector('.gallery-empty');
  const state = { software: 'all', industry: 'all', physics: 'all' };

  function apply() {
    let visible = 0;
    items.forEach(it => {
      const s = (it.dataset.software || '').split(' ');
      const i = (it.dataset.industry || '').split(' ');
      const p = (it.dataset.physics || '').split(' ');
      const ok = (state.software === 'all' || s.includes(state.software))
              && (state.industry === 'all' || i.includes(state.industry))
              && (state.physics === 'all' || p.includes(state.physics));
      it.style.display = ok ? '' : 'none';
      if (ok) visible++;
    });
    if (empty) empty.hidden = visible > 0;
  }

  buttons.forEach(btn => {
    btn.addEventListener('click', () => {
      const f = btn.dataset.filter;
      const v = btn.dataset.value;
      state[f] = v;
      document.querySelectorAll(`.filter-btn[data-filter="${f}"]`).forEach(b => b.classList.toggle('active', b === btn));
      apply();
    });
  });
  apply();
})();
</script>
